void smain(uint entra){
	uint sai = decide();

	//Checando se a página está suja.
	if((memV[sai].controle & ALTERADO) != 0){
		n_dirty_pages++;
	}

	//Zera os bits da página que saiu.
	memV[sai].controle = 0;

}
